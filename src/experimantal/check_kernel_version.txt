// Проверка версии ядра в рантайме
void check_kernel_version() {
    struct utsname sysinfo;
    uname(&sysinfo);
    printf("ncpp: работает на ядре %s\n", sysinfo.release);

    #if defined(__GNUC__)
        printf("Собран с GCC %d.%d.%d\n", 
               __GNUC__, __GNUC_MINOR__, __GNUC_PATCHLEVEL__);
	#elif defined(__clang__)
		printf("Собран с Clang %d.%d.%d\n", 
               __clang_major__, __clang_minor__, __clang_patchlevel__);
    #endif
}
